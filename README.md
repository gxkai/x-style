```
npm install
```

### Compiles and hot-reloads for development
```
npm run serve
```

### Compiles and minifies for production
```
npm run build
```

```
npm run test
```

### Customize configuration

```vue
import xui from 'x-style'
import 'x-style/lib/x-style.css'

```
